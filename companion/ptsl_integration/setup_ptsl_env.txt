# PTSL Integration Setup Script
# Run this to set up the Python environment for PTSL integration

# 1. Create virtual environment
python -m venv venv_ptsl

# 2. Activate virtual environment
# Windows PowerShell:
# .\venv_ptsl\Scripts\Activate.ps1

# Windows CMD:
 .\venv_ptsl\Scripts\activate.bat

# 3. Install required packages
pip install grpcio grpcio-tools requests protobuf

# 4. Generate Python protobuf files from PTSL.proto
python -m grpc_tools.protoc `
    -IC:\Users\Ludenbold\Desktop\Master_Thesis\Implementation\Avid\PTSL_SDK_CPP.2025.06.0.1178210\Source `
    --python_out=. `
    --grpc_python_out=. `
    C:\Users\Ludenbold\Desktop\Master_Thesis\Implementation\Avid\PTSL_SDK_CPP.2025.06.0.1178210\Source\PTSL.proto

# 5. Verify generated files exist
Write-Host "`nChecking generated files:"
if (Test-Path "PTSL_pb2.py") {
    Write-Host "✅ PTSL_pb2.py generated"
} else {
    Write-Host "❌ PTSL_pb2.py missing"
}

if (Test-Path "PTSL_pb2_grpc.py") {
    Write-Host "✅ PTSL_pb2_grpc.py generated"
} else {
    Write-Host "❌ PTSL_pb2_grpc.py missing"
}

Write-Host "`n✅ Setup complete!"
Write-Host "Virtual environment: venv_ptsl"
Write-Host "`nTo activate later, run:"
Write-Host "  .\venv_ptsl\Scripts\Activate.ps1"
